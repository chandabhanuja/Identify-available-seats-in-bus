#define sensor4 A3;

#define sensor6 A5; 
float val1;

{
  lcd.begin(16, 2);
  pinMode(sensor1,INPUT);
  pinMode(sensor2,INPUT);
  pinMode(sensor3,INPUT);
  pinMode(sensor4,INPUT);
  pinMode(sensor5,INPUT);
  pinMode(sensor6,INPUT);
  pinMode(val1,OUTPUT);
  pinMode(val2,OUTPUT);
  pinMode(val3,OUTPUT);
  pinMode(val4,OUTPUT);
  pinMode(val5,OUTPUT);
  pinMode(val6,OUTPUT);
}

  if(val5>16)
